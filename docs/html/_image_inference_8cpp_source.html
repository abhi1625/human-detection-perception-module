64 vector<Mat> ImageInference::runForwardPass(const Mat &inferImage) {

67  cv::dnn::blobFromImage(inferImage, blob, 1/255.0,

80 vector<cv::String> ImageInference::feedOutputNames(const cv::dnn::Net& net) {
81  static vector<cv::String> outputLayerStrings;
82  if (outputLayerStrings.empty()) {
83  // Get the indices of the output layers,
84  // i.e. the layers with unconnected outputs
85  vector<int> outLayers = net.getUnconnectedOutLayers();
86  // Get the names of all the layers in the network
87  vector<cv::String> layersNames = net.getLayerNames();
88  // Get the names of the output layers in names
89  outputLayerStrings.resize(outLayers.size());
90  for (size_t i = 0; i < outLayers.size(); ++i) {
91  outputLayerStrings[i] = layersNames[outLayers[i] - 1];
92  }
93  }
94  return outputLayerStrings;
95 }

97 void ImageInference::applyNMS(const Mat& inferImage, const vector<Mat>& finalLayer) {
98  // Stich the final layer outputs for bounding box
99  // coordinates and gather the labels. Finally apply
100  // Non-Maximal Supression(NMS) to suppress boxes with
101  // low confidences.
102  for (size_t i = 0; i < finalLayer.size(); ++i) {
103  float* data = reinterpret_cast<float*>(finalLayer[i].data);
104  for (int j = 0; j < finalLayer[i].rows; ++j,
105  data += finalLayer[i].cols) {
106  Mat scores = finalLayer[i].row(j).colRange(5, finalLayer[i].cols);
107  cv::Point labelLocations;
108  double confidences;
109  // Get the value and location of the maximum score
110  cv::minMaxLoc(scores, 0, &confidences, 0, &labelLocations);
111  if (confidences > confidenceThresh) {
112  int centerX = static_cast<int>(data[0] * inferImage.cols);
113  int centerY = static_cast<int>(data[1] * inferImage.rows);
114  int width = static_cast<int>(data[2] * inferImage.cols);
115  int height = static_cast<int>(data[3] * inferImage.rows);
116  int left = centerX - width / 2;
117  int top = centerY - height / 2;
118  if (labelLocations.x == 0) {
119  labels.push_back(labelLocations.x);
120  labelConfidences.push_back(static_cast<float>(confidences));
121  boxCords.push_back(cv::Rect(left, top, width, height));
122  }
123  }
124  }
125  }
126 
127  // perform non-maximal suppression
128  vector<int> indices;
129  cv::dnn::NMSBoxes(boxCords, labelConfidences,
130  confidenceThresh, nmsThresh, indices);
131 }
